Write your secure APIs

Simply put your code in "my-apis" folder, and call the public sub-routines from anywhere in the world, via a secure pipeline.


// my-apis/edison.js

var groveSensor = require('jsupm_grove');

var light = new groveSensor.GroveLight(0);

exports.lightStatus = function(cb) {
  cb(null, {
    'lux': light.value()
  });
}
