    $ cat faas-agent.sh

#!/bin/sh

export NODE_PATH=/usr/lib/node_modules
export FAAS_API_KEY=your key
export FAAS_API_SECRET=your secret
export FAAS_IX_API_DIR=my-apis
./node_modules/faas/bin/faas-ix-server
